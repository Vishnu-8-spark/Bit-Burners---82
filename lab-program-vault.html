1. Stack Implementation using Arrays (C Program)

Concept: Demonstrates LIFO structure and stack operations (push, pop, display).

#include 
#define SIZE 5
int stack[SIZE], top = -1;

void push(int val) {
  if (top == SIZE - 1) printf("Stack Overflow\\n");
  else stack[++top] = val;
}

void pop() {
  if (top == -1) printf("Stack Underflow\\n");
  else printf("Popped: %d\\n", stack[top--]);
}

int main() {
  push(10); push(20); pop();
  return 0;
}
